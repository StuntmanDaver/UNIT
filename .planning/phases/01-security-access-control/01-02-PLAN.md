    import { useAuth } from '@/lib/AuthContext';

    export default function LandlordGuard() {
      const { user, isLoadingAuth, isLandlord } = useAuth();

      if (isLoadingAuth) {
        return (
          <div className="fixed inset-0 flex items-center justify-center bg-brand-navy">

    }

    **Create `src/lib/PropertyContext.jsx`:**

    ```jsx
    import { createContext, useContext, useState, useCallback } from 'react';
    import { useQueryClient } from '@tanstack/react-query';

    const PropertyContext = createContext(null);

    export function PropertyProvider({ children }) {
      const queryClient = useQueryClient();
      const [activePropertyId, setActivePropertyId] = useState(
        () => localStorage.getItem('active_property_id') ?? null
      );

      const switchProperty = useCallback((propertyId) => {
        localStorage.setItem('active_property_id', propertyId);
        setActivePropertyId(propertyId);
        queryClient.invalidateQueries();
      }, [queryClient]);

      return (
        <PropertyContext.Provider value={{ activePropertyId, switchProperty }}>
          {children}
        </PropertyContext.Provider>
      );
    }

    export const useProperty = () => {
      const ctx = useContext(PropertyContext);
      if (!ctx) throw new Error('useProperty must be used within PropertyProvider');
      return ctx;
    };
    ```

    Per D-10: active property persisted to localStorage key `active_property_id`.
    Per D-11: `switchProperty` calls `queryClient.invalidateQueries()` with no arguments to invalidate all queries atomically.
    Per D-08: switching stays on same page (data reloads in place — handled by query invalidation, not navigation).

    Create the `src/components/guards/` directory if it does not exist.
